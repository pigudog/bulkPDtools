xgb_cv_bayes <- function(eta, max.depth, min_child_weight, subsample) {
  cv <- xgb.cv(
    params = list(
      booster = "gbtree",
      eta = eta,
      max_depth = max.depth,
      min_child_weight = min_child_weight,
      subsample = subsample,
      colsample_bytree = 0.6,
      lambda = 1,
      alpha = 0,
      objective = "binary:logistic",
      eval_metric = "auc"
    ),
    data = dtrain,
    #减少循环次数以作示例，可增大（到100）
    nround = 10,
    folds = cv_folds,
    prediction = TRUE,
    showsd = TRUE,
    early.stop.round = 5,
    maximize = TRUE,
    verbose = 0
  )
  list(
    Score = cv$evaluation_log[, max(test_auc_mean)], Pred = cv$pred
  )
}
